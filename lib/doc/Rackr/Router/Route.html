def initialize(path, endpoint, befores: [], afters: [])
  @path = path
  @splitted_path = @path.split('/')
  @endpoint = endpoint
  @params = fetch_params
  @has_params = @params != []
  @befores = befores
  @has_befores = befores != []
  @afters = afters
  @has_afters = afters != []
  @path_regex = /\A#{path.gsub(/(:\w+)/, '([^/]+)')}\z/
end

def match?(env)
  return env['PATH_INFO'].match?(@path_regex) if @has_params

  env['PATH_INFO'] == @path
end
